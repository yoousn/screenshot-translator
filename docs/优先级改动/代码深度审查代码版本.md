### B1　✅ 快捷键 450ms 共享防抖
**说明：** 两个热键共用一个时间戳，按完一个 450ms 内另一个失灵。改成**每个热键各自一份防抖**。

```rust
// src-tauri/src/hotkeys.rs
use std::collections::HashMap;
use std::sync::Mutex;
use once_cell::sync::Lazy;

static LAST_FIRED: Lazy<Mutex<HashMap<String, Instant>>> =
    Lazy::new(|| Mutex::new(HashMap::new()));
const DEBOUNCE: Duration = Duration::from_millis(450);

fn should_fire(action: &str) -> bool {
    let mut map = LAST_FIRED.lock().unwrap();
    let now = Instant::now();
    match map.get(action) {
        Some(prev) if now.duration_since(*prev) < DEBOUNCE => false,
        _ => { map.insert(action.to_string(), now); true }
    }
}
```

## 二、安全修复

### S1　✅ SSRF / DNS Rebinding —— IP-pinning 专用 transport（已落地，与仓库一致）
**说明：** 不能“把域名换成 IP”发请求（会破坏 TLS/SNI 与证书校验）。正确做法：**域名仍留给 SNI/证书校验，只把 TCP 连接钉到“解析+校验同步完成”的安全 IP**，并禁止隐式代理。

**(1) 解析 + 逐地址校验（`server/security.py`）**
```python
import ipaddress, socket

def _is_blocked_ip(ip: ipaddress._BaseAddress) -> bool:
    return (ip.is_private or ip.is_loopback or ip.is_link_local
            or ip.is_reserved or ip.is_unspecified or ip.is_multicast)

def iter_safe_addresses(host: str, port: int, *, allow_private: bool = False):
    try:
        addr_info = socket.getaddrinfo(host, port, type=socket.SOCK_STREAM, proto=socket.IPPROTO_TCP)
    except OSError as exc:
        raise ValueError("请求地址无法解析") from exc
    for family, socktype, proto, _canon, sockaddr in addr_info:
        ip_str = sockaddr[0].split("%", 1)[0]
        if _is_blocked_ip(ipaddress.ip_address(ip_str)) and not allow_private:
            continue
        yield family, socktype, proto, sockaddr, ip_str

def resolve_safe_ip(host: str, *, allow_private: bool = False) -> str:
    for *_p, ip_str in iter_safe_addresses(host, 0, allow_private=allow_private):
        return ip_str
    raise ValueError("请求地址不合法 (无可用公网 IP)")
```

**(2) IP-pinning 的自定义 transport（`server/safe_transport.py`）**
```python
import socket, sys
from requests.adapters import HTTPAdapter
from urllib3.connection import HTTPConnection, HTTPSConnection
from urllib3.connectionpool import HTTPConnectionPool, HTTPSConnectionPool
from urllib3.exceptions import ConnectTimeoutError, NameResolutionError, NewConnectionError
from urllib3.poolmanager import PoolManager
from urllib3.util.connection import _DEFAULT_TIMEOUT, _set_socket_options  # 注：下面“注意事项”
from security import iter_safe_addresses

def _connect_pinned_address(conn, allow_private):
    # 在 _dns_host（仍是域名）上重新解析+校验，连到被校验的 IP
    safe = list(iter_safe_addresses(conn._dns_host, conn.port, allow_private=allow_private))
    if not safe:
        raise NameResolutionError(conn.host, conn, ValueError("无可用公网 IP"))
    err = None
    for family, socktype, proto, sockaddr, _ip in safe:
        sock = None
        try:
            sock = socket.socket(family, socktype, proto)
            _set_socket_options(sock, conn.socket_options)
            if conn.timeout is not _DEFAULT_TIMEOUT:
                sock.settimeout(conn.timeout)
            if conn.source_address:
                sock.bind(conn.source_address)
            sock.connect(sockaddr)
            return sock
        except OSError as exc:
            err = exc
            if sock is not None:
                sock.close()
    raise err or OSError("no safe address")

def _make_connection_cls(base, allow_private):
    class _Pinned(base):
        def _new_conn(self):
            sock = _connect_pinned_address(self, allow_private)
            sys.audit("http.client.connect", self, self.host, self.port)
            return sock
    return _Pinned

def build_safe_pool_manager(allow_private, **kw):
    class _HttpPool(HTTPConnectionPool):  ConnectionCls = _make_connection_cls(HTTPConnection, allow_private)
    class _HttpsPool(HTTPSConnectionPool): ConnectionCls = _make_connection_cls(HTTPSConnection, allow_private)
    pm = PoolManager(**kw)
    pm.pool_classes_by_scheme = {"http": _HttpPool, "https": _HttpsPool}
    return pm

class SSRFSafeAdapter(HTTPAdapter):
    def __init__(self, *a, allow_private=False, **kw):
        self._allow_private = allow_private
        super().__init__(*a, **kw)
    def init_poolmanager(self, connections, maxsize, block=False, **kw):
        self.poolmanager = build_safe_pool_manager(self._allow_private,
            num_pools=connections, maxsize=maxsize, block=block, **kw)
    def proxy_manager_for(self, *a, **kw):
        raise RuntimeError("SSRF-safe sessions must be used without an outbound proxy.")
```
> 关键：`_dns_host` 始终是域名 → SNI / 证书校验 / Host 头全走域名，**TLS 不会坏**；只有 `_new_conn` 的 TCP 目标被换成校验过的 IP，且解析+校验+连接同一步完成 → TOCTOU 闭合。

**(3) public / relay 两套 session（`server/http_client.py`）**
```python
import requests
from safe_transport import SSRFSafeAdapter

def _make_session(*, allow_private):
    s = requests.Session()
    s.trust_env = False  # 避免隐式读 HTTP(S)_PROXY 绕过 pinning
    adapter = SSRFSafeAdapter(allow_private=allow_private, pool_connections=10, pool_maxsize=20)
    s.mount("http://", adapter); s.mount("https://", adapter)
    return s

_PUBLIC_SESSION = _make_session(allow_private=False)  # Google/Baidu/DeepL
_RELAY_SESSION  = _make_session(allow_private=True)   # 用户自配的 LAN 中转
```

> ⏳ **注意事项（维护点）：** `_DEFAULT_TIMEOUT` / `_set_socket_options` 是 urllib3 的**下划线私有 API**，urllib3 升级（尤其 v1→v2）可能改名或移除。建议二选一：在 `requirements.txt` **钉死 urllib3 版本范围**（如 `urllib3>=2,<3`）；或加一层容错 import：
```python
try:
    from urllib3.util.connection import _DEFAULT_TIMEOUT, _set_socket_options
except ImportError:  # 版本兑变时的兼容兽
    import socket as _s
    _DEFAULT_TIMEOUT = _s._GLOBAL_DEFAULT_TIMEOUT
    def _set_socket_options(sock, options):
        for opt in options or ():
            sock.setsockopt(*opt)
```
